FROM python:3.9-slim
WORKDIR /app

# Copy and install dependencies
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy the model artefacts (scaler + mlruns folder)
COPY models/scaler.joblib models/
COPY mlruns mlruns/

# Copy the API script
COPY serve.py .

EXPOSE 8000

CMD ["python", "serve.py"]
